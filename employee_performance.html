import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# ---- STEP 1: Create Synthetic Dataset (100 Employees) ----

np.random.seed(22)

departments = ["Sales", "IT", "HR", "Finance", "Marketing", "Operations", "Support", "Logistics"]
regions = ["South", "North", "East", "West"]

# Generate dataset
df = pd.DataFrame({
    "Employee_ID": range(1, 101),
    "Department": np.random.choice(departments, size=100, p=[0.20, 0.15, 0.10, 0.10, 0.10, 0.15, 0.10, 0.10]),
    "Region": np.random.choice(regions, size=100),
    "Performance_Score": np.random.randint(60, 100, size=100)
})

print("First 5 rows of dataset:")
print(df.head())

# ---- STEP 2: Count frequency of 'Sales' department ----
sales_count = (df["Department"] == "Sales").sum()
print("\nSales Department Count =", sales_count)

# ---- STEP 3: Create Histogram of Department Distribution ----
plt.figure(figsize=(10, 5))
sns.countplot(data=df, x="Department", palette="viridis")
plt.title("Department Distribution - Employee Performance Dataset")
plt.xticks(rotation=30)
plt.tight_layout()

# Save histogram as PNG
plt.savefig("departments_hist.png")
print("\nSaved: departments_hist.png")

# ---- STEP 4: Export as HTML ----

html_content = f"""
